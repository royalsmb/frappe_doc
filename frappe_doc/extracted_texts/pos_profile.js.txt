frappe.ui.form.on('POS Profile', {
	refresh(frm) {
		// Filter warehouses to include all available companies
		frm.set_query('warehouse', 'profile_warehouses', function() {
			return {
				filters: {
					// Fetch warehouses for all companies
					company: ['in', frappe.db.get_list('Company', {fields: ['name']}).map(c => c.name)]
				}
			};
		});
	}
})
