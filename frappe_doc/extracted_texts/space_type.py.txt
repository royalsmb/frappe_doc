import frappe
from frappe.model.document import Document


class SpaceType(Document):
	def validate(self):
		item_group = frappe.get_doc({
			'doctype': 'Item Group',
			'parent_item_group': 'All Item Groups',
			'item_group_name': self.type_name,
		})
		item_group.save(ignore_permissions=True)
		frappe.db.commit()
