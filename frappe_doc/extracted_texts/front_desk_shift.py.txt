import frappe
from frappe.model.document import Document


class FrontDeskShift(Document):
	def validate(self):	
		user = frappe.session.user
		full_name = frappe.db.get_value("User", user, "full_name")
		self.user = full_name
		
	def after_insert(self):
		self.remaining_balance = self.opening_cash
		self.save()
		frappe.db.commit()
