import frappe
from frappe.model.document import Document


class Space(Document):
	def after_insert(self):
		item = frappe.get_doc({
			'doctype': 'Item',
			'item_code': self.space_name,
			'item_group': self.space_type,
			'is_stock_item': 0,
			'gst_hsn_code': self.custom_hsnsac

		})
		item.save(ignore_permissions=True)
		frappe.db.commit()
