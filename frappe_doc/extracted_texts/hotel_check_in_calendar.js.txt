frappe.views.calendar["Hotel Check In"] = {
    field_map: {
        "start": "start",
        "end": "end",
        "id": "id",
        "title": "title",
        "allDay": "allDay",
        "room": "room"  // Optionally display room if needed
    },
    gantt: false,  // Set to true if you want Gantt chart view as well
    filters: [],  // Ensure this is an empty array
    get_events_method: "jokoor_hotel.api.get_events",  // Update this to your actual path
    get_css_class: function(data) {
        if (data.status === "Completed") {
            return "success";  // Use the 'success' CSS class for Checked Out
        } else if (data.status === "Cancelled") {
            return "danger";  // Use the 'danger' CSS class for Cancelled
        } else if (data.status === "To Check Out") {
            return "warning";  // Use the 'warning' CSS class for Checked In
        } else {
            return "";  // Default class if needed
        }
    },
    onload: function(calendar_view) {
        // Clear any URL filters when the calendar is loaded
        frappe.set_route("List", "Hotel Check In", "Calendar");
    }
};
