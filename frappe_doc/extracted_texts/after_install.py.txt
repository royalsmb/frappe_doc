import frappe

@frappe.whitelist()
def create_order_types():
    
    order_types = ["Restaurant", "Take Away", "Room", "Staff"]
    for ot in order_types:
        if not frappe.db.exists("Order Type", ot):
            doc = frappe.new_doc("Order Type")
            doc.order_type = ot
            doc.insert()
            frappe.db.commit()
    return frappe.session.user
