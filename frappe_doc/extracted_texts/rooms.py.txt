from __future__ import unicode_literals
import frappe
from frappe.model.document import Document

class Rooms(Document):
	def validate(self):
		if not frappe.db.exists("Item", {'custom_room_id': self.room_number}):
			item = frappe.new_doc("Item")
			item.item_code = f'Room - {self.room_number}'
			item.item_name = f'Room - {self.room_number}'
			item.is_stock_item = 0
			item.include_item_in_manufacturing = 0  
			item.item_group = "Rooms"
			item.standard_rate = self.price
			item.custom_room_id = self.room_number
			item.custom_is_room = 1
			if "india_compliance" in frappe.get_installed_apps():
				item.gst_hsn_code = "996311"
			is_purchased = 0
			item.insert()
			frappe.db.commit()


	def on_trash(self):
		item_name = f'Room - {self.room_number}'
		if frappe.db.exists("Item", item_name):
			frappe.delete_doc("Item", item_name)
